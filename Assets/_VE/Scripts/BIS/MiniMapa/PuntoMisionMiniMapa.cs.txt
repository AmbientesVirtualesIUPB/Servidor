using System.Collections.Generic;
using UnityEngine;
public class PuntoMisionMiniMapa : MonoBehaviour
{
    public List<PuntoMision> puntos = new List<PuntoMision>();
    public RectTransform miniMapRect;
    public GameObject prefabIconoMision;

    public float mapMundoRadio = 50f;
    public float bordeMargen = 4f;

    Transform player;

    public static PuntoMisionMiniMapa singleton;

    void Awake()
    {
        if (singleton == null) singleton = this;
        else Destroy(this);
    }

    void Start()
    {
        player = ControlUsuarios.singleton.usuarioLocal.transform;
    }

    void Update()
    {
        if (!player) return;

        for (int i = 0; i < puntos.Count; i++)
        {
            ActualizarPunto(puntos[i]);
        }
    }

    void ActualizarPunto(PuntoMision punto)
    {
        if (!punto.objetivo) return;

        Vector3 offset = punto.objetivo.position - player.position;
        Vector2 offset2D = new Vector2(offset.x, offset.z);

        float distancia = offset2D.magnitude;

        float radioMapa = Mathf.Min(miniMapRect.rect.width, miniMapRect.rect.height) * 0.5f;
        float radioIcono = punto.icono.rect.width * 0.5f;

        Vector2 pos = (offset2D / mapMundoRadio) * radioMapa;

        if (distancia > mapMundoRadio)
        {
            float radioSeguro = radioMapa - radioIcono - bordeMargen;
            pos = pos.normalized * radioSeguro;
        }

        punto.icono.anchoredPosition = pos;
    }

    public void AgregarMision(Transform objetivo)
    {
        GameObject icono = Instantiate(prefabIconoMision, miniMapRect);

        PuntoMision nuevo = new PuntoMision();
        nuevo.objetivo = objetivo;
        nuevo.icono = icono.GetComponent<RectTransform>();

        puntos.Add(nuevo);
    }

    public void EliminarMision(Transform objetivo)
    {
        for (int i = puntos.Count - 1; i >= 0; i--)
        {
            if (puntos[i].objetivo == objetivo)
            {
                Destroy(puntos[i].icono.gameObject);
                puntos.RemoveAt(i);
            }
        }
    }
}
